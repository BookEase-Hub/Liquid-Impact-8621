struct AnimatedScoreRing: View {
    let score: Int
    var size: CGFloat = 120
    var animated: Bool = true
    
    @State private var progress: Double = 0
    @State private var pulse: Double = 1.0
    
    var gradientColors: [Color] {
        if score >= 80 { return [.green, .cyan] }
        if score >= 50 { return [.orange, .green] }
        return [.red, .orange]
    }
    
    var body: some View {
        ZStack {
            Circle()
                .fill(LinearGradient(colors: gradientColors, startPoint: .topLeading, endPoint: .bottomTrailing))
                .frame(width: size + 40, height: size + 40)
                .blur(radius: 30)
                .opacity(0.3)
                .scaleEffect(pulse)
            
            Circle()
                .stroke(Color.white.opacity(0.08), lineWidth: 12)
                .frame(width: size, height: size)
            
            Circle()
                .trim(from: 0, to: progress / 100)
                .stroke(
                    LinearGradient(colors: gradientColors, startPoint: .topLeading, endPoint: .bottomTrailing),
                    style: StrokeStyle(lineWidth: 12, lineCap: .round)
                )
                .frame(width: size, height: size)
                .rotationEffect(.degrees(-90))
            
            VStack(spacing: 2) {
                Text("\(score)")
                    .font(DesignSystem.typography(size: 36, weight: .black))
                    .foregroundStyle(.white)
                Text("Impact Score")
                    .font(DesignSystem.typography(size: 11, weight: .semibold))
                    .foregroundStyle(.white.opacity(0.6))
            }
        }
        .onAppear {
            guard animated else { progress = Double(score); return }
            withAnimation(.easeInOut(duration: 1.2)) {
                progress = Double(score)
            }
            withAnimation(.easeInOut(duration: 1.0).repeatForever(autoreverses: true).delay(1.5)) {
                pulse = 1.05
            }
        }
    }
}

struct ImpactScoreView: View {
    let score: Int
    var size: ScoreSize = .medium
    var animated: Bool = true
    
    @State private var displayScore: Int = 0
    
    enum ScoreSize {
        case small, medium, large, extraLarge
        
        var dimension: CGFloat {
            switch self {
            case .small: return 64
            case .medium: return 96
            case .large: return 128
            case .extraLarge: return 160
            }
        }
        
        var fontSize: CGFloat {
            switch self {
            case .small: return 24
            case .medium: return 36
            case .large: return 48
            case .extraLarge: return 60
            }
        }
    }
    
    var scoreColor: Color {
        if score >= 80 { return .green }
        if score >= 50 { return .orange }
        return .red
    }
    
    var gradientColors: [Color] {
        if score >= 80 { return [.green, .cyan] }
        if score >= 50 { return [.orange, .green] }
        return [.red, .orange]
    }
    
    var body: some View {
        ZStack {
            Circle()
                .fill(
                    LinearGradient(
                        colors: gradientColors,
                        startPoint: .topLeading,
                        endPoint: .bottomTrailing
                    )
                )
                .frame(width: size.dimension, height: size.dimension)
                .shadow(color: scoreColor.opacity(0.4), radius: 16, x: 0, y: 8)
            
            Text("\(displayScore)")
                .font(.system(size: size.fontSize, weight: .black, design: .rounded))
                .foregroundStyle(.white)
        }
        .onAppear {
            if animated {
                withAnimation(.spring(response: 0.8, dampingFraction: 0.7)) {
                    displayScore = score
                }
            } else {
                displayScore = score
            }
        }
        .onChange(of: score) { oldValue, newValue in
            withAnimation(.spring(response: 0.8, dampingFraction: 0.7)) {
                displayScore = newValue
            }
        }
    }
}

struct ScanCornerShape: Shape {
    func path(in rect: CGRect) -> Path {
        var path = Path()
        let cornerLength: CGFloat = 40
        let lineWidth: CGFloat = 3
        
        path.move(to: CGPoint(x: rect.minX, y: rect.minY + cornerLength))
        path.addLine(to: CGPoint(x: rect.minX, y: rect.minY + lineWidth/2))
        path.addLine(to: CGPoint(x: rect.minX + cornerLength, y: rect.minY + lineWidth/2))
        
        path.move(to: CGPoint(x: rect.maxX - cornerLength, y: rect.minY + lineWidth/2))
        path.addLine(to: CGPoint(x: rect.maxX, y: rect.minY + lineWidth/2))
        path.addLine(to: CGPoint(x: rect.maxX, y: rect.minY + cornerLength))
        
        path.move(to: CGPoint(x: rect.maxX, y: rect.maxY - cornerLength))
        path.addLine(to: CGPoint(x: rect.maxX, y: rect.maxY - lineWidth/2))
        path.addLine(to: CGPoint(x: rect.maxX - cornerLength, y: rect.maxY - lineWidth/2))
        
        path.move(to: CGPoint(x: rect.minX + cornerLength, y: rect.maxY - lineWidth/2))
        path.addLine(to: CGPoint(x: rect.minX, y: rect.maxY - lineWidth/2))
        path.addLine(to: CGPoint(x: rect.minX, y: rect.maxY - cornerLength))
        
        return path
    }
}

// MARK: - Color Hex Parser

extension Color {
    init?(hex: String, fallback: Color = .clear) {
        let cleanHex = hex.trimmingCharacters(in: .whitespacesAndNewlines)
            .replacingOccurrences(of: "#", with: "")
        
        guard cleanHex.count == 6 || cleanHex.count == 8 else {
            self = fallback
            return
        }
        
        var int: UInt64 = 0
        guard Scanner(string: cleanHex).scanHexInt64(&int) else {
            self = fallback
            return
        }
        
        let a, r, g, b: UInt64
        switch cleanHex.count {
        case 6: (a, r, g, b) = (255, int >> 16, int >> 8 & 0xFF, int & 0xFF)
        case 8: (a, r, g, b) = (int >> 24, int >> 16 & 0xFF, int >> 8 & 0xFF, int & 0xFF)
        default:
            self = fallback
            return
        }
        
        self.init(
            .sRGB,
            red: Double(r) / 255,
            green: Double(g) / 255,
            blue: Double(b) / 255,
            opacity: Double(a) / 255
        )
    }
    
    static func hex(_ hex: String, fallback: Color = .clear) -> Color {
        Color(hex: hex, fallback: fallback) ?? fallback
    }
}

@Observable
class ThemeManager {
    static let shared = ThemeManager()

    // Default to dark mode
    var isDarkMode: Bool = true

    // MARK: - Colors

    // Primary Backgrounds
    var backgroundPrimary: Color {
        isDarkMode ? Color(hex: "0A0A0F") : Color(hex: "FAFAFA")
    }

    var backgroundSecondary: Color {
        isDarkMode ? Color(hex: "151520") : Color(hex: "F0F0F5")
    }

    var backgroundTertiary: Color {
        isDarkMode ? Color(hex: "1E1E2E") : Color(hex: "E8E8F0")
    }

    var surface: Color {
        isDarkMode ? Color(hex: "2A2A3C") : Color(hex: "FFFFFF")
    }

    // Typography Colors
    var textPrimary: Color {
        isDarkMode ? Color(hex: "FFFFFF") : Color(hex: "1A1A2E")
    }

    var textSecondary: Color {
        isDarkMode ? Color(hex: "A0A0B0") : Color(hex: "6B6B80")
    }

    var textTertiary: Color {
        isDarkMode ? Color(hex: "6B6B80") : Color(hex: "9090A0")
    }

    // Accent Colors
    var accentCyan: Color { Color(hex: "00B4D8") }
    var accentPurple: Color { Color(hex: "7B2CBF") }
    var accentPink: Color { Color(hex: "FF6B9D") }
    var accentGreen: Color { Color(hex: "00C853") }
    var accentOrange: Color { Color(hex: "FF9800") }
    var accentRed: Color { Color(hex: "FF3D00") }
    var accentYellow: Color { Color(hex: "FFD600") }
    var accentBlue: Color { Color(hex: "2196F3") }

    // Gradients
    var gradientPrimary: LinearGradient {
        LinearGradient(
            colors: [accentCyan, accentPurple],
            startPoint: .leading,
            endPoint: .trailing
        )
    }

    var gradientScan: LinearGradient {
        LinearGradient(
            colors: [Color(hex: "00B4D8"), Color(hex: "7B2CBF")],
            startPoint: .leading,
            endPoint: .trailing
        )
    }

    var gradientScore: LinearGradient {
        LinearGradient(
            colors: [accentPurple, accentGreen],
            startPoint: .leading,
            endPoint: .trailing
        )
    }

    // Glassmorphism
    var glassBackground: Color {
        isDarkMode ? Color.white.opacity(0.05) : Color.black.opacity(0.03)
    }

    var glassBorder: Color {
        isDarkMode ? Color.white.opacity(0.12) : Color.black.opacity(0.08)
    }

    var glassShadow: Color {
        isDarkMode ? accentCyan.opacity(0.25) : accentBlue.opacity(0.15)
    }

    // Semantic aliases
    var cardBackground: Color { backgroundSecondary }
    var dividerColor: Color { textTertiary.opacity(0.3) }
    var placeholderColor: Color { textSecondary.opacity(0.6) }

    // MARK: - Methods

    func toggleTheme() {
        withAnimation(.spring(response: 0.3, dampingFraction: 0.8)) {
            isDarkMode.toggle()
        }
    }

    func setDarkMode(_ enabled: Bool) {
        withAnimation(.spring(response: 0.3, dampingFraction: 0.8)) {
            isDarkMode = enabled
        }
    }
}

// MARK: - Color Extension

extension Color {
    init(hex: String) {
        let hex = hex.trimmingCharacters(in: CharacterSet.alphanumerics.inverted)
        var int: UInt64 = 0
        Scanner(string: hex).scanHexInt64(&int)
        let a, r, g, b: UInt64
        switch hex.count {
        case 3:
            (a, r, g, b) = (255, (int >> 8) * 17, (int >> 4 & 0xF) * 17, (int & 0xF) * 17)
        case 6:
            (a, r, g, b) = (255, int >> 16, int >> 8 & 0xFF, int & 0xFF)
        case 8:
            (a, r, g, b) = (int >> 24, int >> 16 & 0xFF, int >> 8 & 0xFF, int & 0xFF)
        default:
            (a, r, g, b) = (255, 0, 0, 0)
        }
        self.init(
            .sRGB,
            red: Double(r) / 255,
            green: Double(g) / 255,
            blue: Double(b) / 255,
            opacity: Double(a) / 255
        )
    }
}

class HapticsService {
    static let shared = HapticsService()
    private var engine: CHHapticEngine?
    private var isPrepared = false
    
    private init() {
        prepareEngine()
    }
    
    private func prepareEngine() {
        guard CHHapticEngine.capabilitiesForHardware().supportsHaptics else { return }
        do {
            engine = try CHHapticEngine()
            engine?.playsHapticsOnly = true
            try engine?.start()
            isPrepared = true
        } catch {
            print("Haptics engine failed: \(error)")
        }
    }
    
    func impact(style: UIImpactFeedbackGenerator.FeedbackStyle = .medium) {
        UIImpactFeedbackGenerator(style: style).impactOccurred()
    }
    
    func notification(type: UINotificationFeedbackGenerator.FeedbackType) {
        UINotificationFeedbackGenerator().notificationOccurred(type)
    }
    
    func playCustomPattern(intensity: Float = 1.0, sharpness: Float = 0.5) {
        guard isPrepared, let engine else { return }
        
        do {
            let events: [CHHapticEvent] = [
                .init(eventType: .hapticContinuous, parameters: [
                    .init(parameterID: .hapticIntensity, value: intensity),
                    .init(parameterID: .hapticSharpness, value: sharpness)
                ], relativeTime: 0, duration: 0.1)
            ]
            let pattern = try CHHapticPattern(events: events, parameters: [])
            let player = try engine.makePlayer(with: pattern)
            try player.start(atTime: CHHapticTimeImmediate)
        } catch {
            print("Custom haptic failed: \(error)")
        }
    }
    
    func scanSequence(score: Int) {
        impact(style: .light)
        
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.1) {
            self.impact(style: .medium)
        }
        
        DispatchQueue.main.asyncAfter(deadline: .now() + 1.5) {
            if score >= 80 {
                self.notification(type: .success)
                self.playCustomPattern(intensity: 0.8, sharpness: 0.3)
            } else if score >= 50 {
                self.notification(type: .warning)
            } else {
                self.notification(type: .error)
                self.playCustomPattern(intensity: 0.4, sharpness: 0.8)
            }
        }
    }
    
    deinit {
        engine?.stop()
    }
}

struct NotificationManager {
    static func requestAuthorization() {
        UNUserNotificationCenter.current().requestAuthorization(options: [.alert, .badge, .sound]) { granted, _ in
            print("Notification permission: \(granted)")
        }
    }
    
    static func schedule(title: String, body: String, delay: TimeInterval = 1) {
        let content = UNMutableNotificationContent()
        content.title = title
        content.body = body
        content.sound = .default
        
        let trigger = UNTimeIntervalNotificationTrigger(timeInterval: delay, repeats: false)
        let request = UNNotificationRequest(identifier: UUID().uuidString, content: content, trigger: trigger)
        
        UNUserNotificationCenter.current().add(request)
    }
}

import Foundation

@Observable
class StreakManager {
    var currentStreak: Int = 0
    var bestStreak: Int = 0
    var lastScanDate: Date?
    var milestones: [Int] = [3, 7, 14, 30, 60, 90]
    
    func updateStreak() {
        let today = Calendar.current.startOfDay(for: Date())
        guard let last = lastScanDate else {
            currentStreak = 1
            lastScanDate = today
            return
        }
        let lastDay = Calendar.current.startOfDay(for: last)
        if today == lastDay { return }
        
        let diff = Calendar.current.dateComponents([.day], from: lastDay, to: today).day ?? 0
        if diff == 1 {
            currentStreak += 1
            if currentStreak > bestStreak { bestStreak = currentStreak }
            checkMilestones()
        } else {
            currentStreak = 1
        }
        lastScanDate = today
    }
    
    private func checkMilestones() {
        if milestones.contains(currentStreak) {
            NotificationManager.schedule(
                title: "🔥 \(currentStreak) Day Streak!",
                body: "Keep your hydration streak alive!"
            )
        }
    }
}

                cta: sodaCount >= 3 ? "Unlock habit protocol" : nil,

    }

class AppState {
    var scanStatus: ScanFlowStatus = .idle
    var latestScanResult: ScanResultSummary?
    var subscriptionStatus: SubscriptionStatus = .free
    var hasCompletedOnboarding: Bool = false
    var streakDays: Int = 7
    var totalScans: Int = 24

    var weeklyTrends: [WeeklyTrend] = []
    var missions: [DailyMission] = []
    var dailyScanCount: Int = 0
    var lastScanDate: Date?
    
    var currentUserStage: UserStage {
        if streakDays <= 7 { return .week1 }
        if streakDays <= 21 { return .week2 }
        return .veteran
    }
    
    var canScan: Bool {
        subscriptionStatus == .free ? (dailyScanCount < 3) : true
    }
    
    var sodaCount: Int {
        recentScans.filter { $0.status == .risky || $0.status == .damaging }.count
    }
    
    init() {
        loadMockData()
    }
    
    func setScanStatus(_ status: ScanFlowStatus) {
        scanStatus = status
    }
    
    func setLatestScan(_ result: ScanResultSummary) {
        latestScanResult = result
        scanStatus = .result
        recentScans.insert(result, at: 0)
        totalScans += 1
        dailyScanCount += 1
        updateStreak()
    }
    
    func updateSubscription(_ status: SubscriptionStatus) {
        subscriptionStatus = status
    }
    
    func resetScanFlow() {
        scanStatus = .idle
    }
    
    func completeOnboarding() {
        hasCompletedOnboarding = true
    }
    
    func completeMission(id: UUID) {
        if let index = missions.firstIndex(where: { $0.id == id }) {
            missions[index].completed = true
            missions[index].progress = missions[index].target
        }
    }
    
    private func updateStreak() {
        let today = Calendar.current.startOfDay(for: Date())
        let yesterday = Calendar.current.date(byAdding: .day, value: -1, to: today)!
        let last = Calendar.current.startOfDay(for: lastScanDate ?? .distantPast)
        
        if last == yesterday {
            streakDays += 1
        } else if last != today {
            streakDays = 1
        }
        lastScanDate = today
    }
    
    private func loadMockData() {
        recentScans = ScanService.shared.mockRecentScans()
        weeklyTrends = ScanService.shared.mockWeeklyTrends()
        missions = [
            DailyMission(id: UUID(), title: "Scan 3 drinks today", description: "+10 Impact", progress: 1, target: 3, completed: false, type: "scan", impact: 10, expiresAt: Date().addingTimeInterval(86400)),
            DailyMission(id: UUID(), title: "Replace one sugary drink", description: "Swap for water/tea", progress: 0, target: 1, completed: false, type: "sugar", impact: 15, expiresAt: Date().addingTimeInterval(86400))
        ]
    }

struct ProgressDashboardView: View {

        case week = "Week"
        case month = "Month"
        case year = "Year"

        guard !weeklyTrends.isEmpty else { return 0 }
        return weeklyTrends.reduce(0) { $0 + $1.averageScore } / Double(weeklyTrends.count)
    }
    
    var totalDrinks: Int {
        weeklyTrends.reduce(0) { $0 + $1.totalDrinks }

struct FlowLayout: Layout {
    var spacing: CGFloat = 8
    
    func sizeThatFits(proposal: ProposedViewSize, subviews: Subviews, cache: inout ()) -> CGSize {
        let result = FlowResult(in: proposal.width ?? 0, subviews: subviews, spacing: spacing)
        return result.size
    }
    
    func placeSubviews(in bounds: CGRect, proposal: ProposedViewSize, subviews: Subviews, cache: inout ()) {
        let result = FlowResult(in: bounds.width, subviews: subviews, spacing: spacing)
        for (index, subview) in subviews.enumerated() {
            subview.place(at: CGPoint(x: bounds.minX + result.positions[index].x, y: bounds.minY + result.positions[index].y), proposal: .unspecified)
        }
    }
    
    struct FlowResult {
        var size: CGSize = .zero
        var positions: [CGPoint] = []
        
        init(in maxWidth: CGFloat, subviews: Subviews, spacing: CGFloat) {
            var x: CGFloat = 0
            var y: CGFloat = 0
            var lineHeight: CGFloat = 0
            
            for subview in subviews {
                let size = subview.sizeThatFits(.unspecified)
                if x + size.width > maxWidth && x > 0 {
                    x = 0
                    y += lineHeight + spacing
                    lineHeight = 0
                }
                positions.append(CGPoint(x: x, y: y))
                lineHeight = max(lineHeight, size.height)
                x += size.width + spacing
            }
            
            self.size = CGSize(width: maxWidth, height: y + lineHeight)
        }
    }
}

        HapticsService.shared.impact(style: .heavy)
        isScanning = true
        step = 0
        
        Task {
            do {
                let result = try await ScanService.shared.processScan(imageUrl: "")
                await MainActor.run {
                    scanResult = result
                    appState.setLatestScan(result)
                    isScanning = false
                    showResult = true
                }
            } catch {
                await MainActor.run {
                    isScanning = false
                }
            }
        }
    }

// Middleware for soft deletes
prisma.$use(async (params, next) => {
  if (params.action === 'findUnique' || params.action === 'findFirst') {
    // Change to findFirst - you cannot filter by anything except ID / unique with findUnique
    params.action = 'findFirst';
    // Add 'deletedAt' filter
    params.args.where['deletedAt'] = null;
  }

  if (params.action === 'findMany') {
    // Find many queries
    if (params.args.where) {
      if (params.args.where.deletedAt == undefined) {
        // Exclude deleted records if they have not been explicitly requested
        params.args.where['deletedAt'] = null;
      }
    } else {
      params.args['where'] = { deletedAt: null };
    }
  }

  if (params.action == 'update') {
    // Change to updateMany - you cannot filter by anything except ID / unique with update
    params.action = 'updateMany';
    // Add 'deletedAt' filter
    params.args.where['deletedAt'] = null;
  }

  if (params.action == 'updateMany') {
    if (params.args.where != undefined) {
      params.args.where['deletedAt'] = null;
    } else {
      params.args['where'] = { deletedAt: null };
    }
  }

  if (params.action == 'delete') {
    // Delete queries
    // Change action to an update
    params.action = 'update';
    params.args['data'] = { deletedAt: new Date() };
  }

  if (params.action == 'deleteMany') {
    // Delete many queries
    params.action = 'updateMany';
    if (params.args.data != undefined) {
      params.args.data['deletedAt'] = new Date();
    } else {
      params.args['data'] = { deletedAt: new Date() };
    }
  }

  return next(params);
});

export const authenticate = async (
  req: AuthRequest,
  res: Response,
  next: NextFunction
): Promise<void> => {
  try {
    const authHeader = req.headers.authorization;

    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      res.status(401).json({ error: 'No token provided' });
      return;
    }

    const token = authHeader.split(' ')[1];

    // Verify JWT
    const decoded = jwt.verify(token, process.env.JWT_SECRET || 'secret') as {
      userId: string;
      email: string;
    };

    // Check if user exists and is active
    const user = await prisma.user.findUnique({
      where: { id: decoded.userId },
      include: {
        subscription: true
      }
    });

    if (!user || user.deletedAt) {
      res.status(401).json({ error: 'User not found' });
      return;
    }

    req.user = {
      id: user.id,
      email: user.email,
      subscriptionTier: user.subscription?.tier || 'FREE'
    };

    next();
  } catch (error) {
    if (error instanceof jwt.TokenExpiredError) {
      res.status(401).json({ error: 'Token expired' });
    } else if (error instanceof jwt.JsonWebTokenError) {
      res.status(401).json({ error: 'Invalid token' });
    } else {
      console.error('Auth error:', error);
      res.status(500).json({ error: 'Authentication failed' });
    }
  }
};

export const requireSubscription = (tier: string) => {
  return (req: AuthRequest, res: Response, next: NextFunction) => {
    if (!req.user) {
      res.status(401).json({ error: 'Authentication required' });
      return;
    }

    const tierHierarchy = {
      'FREE': 0,
      'STARTER': 1,
      'PRO': 2,
      'ELITE': 3,
      'FAMILY': 3
    };

    const userTier = req.user.subscriptionTier.toUpperCase();
    const requiredTier = tier.toUpperCase();

    if (tierHierarchy[userTier] < tierHierarchy[requiredTier]) {
      res.status(403).json({
        error: 'Subscription required',
        requiredTier: tier,
        currentTier: req.user.subscriptionTier
      });
      return;
    }

    next();
  };
};

export const errorHandler = (
  error: Error,
  req: Request,
  res: Response,
  next: NextFunction
): void => {
  console.error('Error:', error);

  if (error instanceof ZodError) {
    res.status(400).json({
      error: 'Validation error',
      details: error.errors
    });
    return;
  }

  // Handle specific error types
  if (error.name === 'PrismaClientKnownRequestError') {
    const prismaError = error as any;

    // Unique constraint violation
    if (prismaError.code === 'P2002') {
      res.status(409).json({
        error: 'Duplicate entry',
        field: prismaError.meta?.target?.[0]
      });
      return;
    }

    // Record not found
    if (prismaError.code === 'P2025') {
      res.status(404).json({ error: 'Record not found' });
      return;
    }
  }

  // Default error
  res.status(500).json({
    error: 'Internal server error',
    message: process.env.NODE_ENV === 'development' ? error.message : undefined
  });
};

// POST /api/auth/refresh - Refresh access token
router.post('/refresh', async (req: Request, res: Response) => {
  try {
    const data = RefreshTokenSchema.parse(req.body);

    // Verify refresh token
    const decoded = jwt.verify(
      data.refreshToken,
      process.env.REFRESH_TOKEN_SECRET || 'refresh_secret'
    ) as { userId: string };

    // Check session
    const session = await prisma.session.findFirst({
      where: {
        userId: decoded.userId,
        refreshToken: data.refreshToken,
        expiresAt: { gt: new Date() }
      }
    });

    if (!session) {
      res.status(401).json({ error: 'Invalid refresh token' });
      return;
    }

    // Get user
    const user = await userService.getUserById(decoded.userId);
    if (!user) {
      res.status(401).json({ error: 'User not found' });
      return;
    }

    // Generate new tokens
    const tokens = generateTokens(user.id, user.email);

    // Update session
    await prisma.session.update({
      where: { id: session.id },
      data: {
        token: tokens.accessToken,
        refreshToken: tokens.refreshToken,
        expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000) // 7 days
      }
    });

    res.json({
      success: true,
      data: tokens
    });
  } catch (error) {
    console.error('Refresh token error:', error);

    if (error instanceof jwt.TokenExpiredError) {
      res.status(401).json({ error: 'Refresh token expired' });
    } else if (error instanceof jwt.JsonWebTokenError) {
      res.status(401).json({ error: 'Invalid refresh token' });
    } else {
      res.status(500).json({ error: 'Failed to refresh token' });
    }
  }
});

// Helper functions
function generateTokens(userId: string, email: string): {
  accessToken: string;
  refreshToken: string;
  expiresIn: number;
} {
  const accessToken = jwt.sign(
    { userId, email },
    process.env.JWT_SECRET || 'secret',
    { expiresIn: process.env.JWT_EXPIRES_IN || '1h' }
  );

  const refreshToken = jwt.sign(
    { userId },
    process.env.REFRESH_TOKEN_SECRET || 'refresh_secret',
    { expiresIn: process.env.REFRESH_TOKEN_EXPIRES_IN || '7d' }
  );

  return {
    accessToken,
    refreshToken,
    expiresIn: 3600 // 1 hour in seconds
  };
}

async function createSession(userId: string, refreshToken: string): Promise<void> {
  const expiresAt = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000); // 7 days

  await prisma.session.create({
    data: {
      id: uuidv4(),
      userId,
      token: '', // Will be set when access token is generated
      refreshToken,
      expiresAt
    }
  });
}

  async getUserAnalytics(userId: string, days: number = 30): Promise<{
    dailyScans: Array<{ date: string; count: number }>;
    missionCompletion: number;
    averageImpact: number;
    retentionRate: number;
    topCategories: Array<{ category: string; count: number }>;
    peakHours: Array<{ hour: number; count: number }>;
  }> {
    const startDate = new Date();
    startDate.setDate(startDate.getDate() - days);

    const [scans, missions, allScans, events] = await Promise.all([
      prisma.scan.findMany({
        where: {
          userId,
          scannedAt: { gte: startDate }
        },
        select: {
          scannedAt: true,
          impactScore: true,
          category: true
        }
      }),
      prisma.mission.findMany({
        where: {
          userId,
          completed: true,
          completedAt: { gte: startDate }
        }
      }),
      prisma.scan.findMany({
        where: { userId },
        select: { scannedAt: true, category: true }
      }),
      prisma.analyticsEvent.findMany({
        where: {
          userId,
          eventType: 'APP_OPEN',
          timestamp: { gte: startDate }
        },
        select: { timestamp: true }
      })
    ]);

    // Daily scans
    const dailyScansMap: Record<string, number> = {};
    scans.forEach(scan => {
      const date = scan.scannedAt.toISOString().split('T')[0];
      dailyScansMap[date] = (dailyScansMap[date] || 0) + 1;
    });

    const dailyScans = Object.entries(dailyScansMap)
      .map(([date, count]) => ({ date, count }))
      .sort((a, b) => a.date.localeCompare(b.date));

    // Mission completion rate
    const missionCompletion = missions.length > 0
      ? (missions.filter(m => m.completed).length / missions.length) * 100
      : 0;

    // Average impact score
    const averageImpact = scans.length > 0
      ? scans.reduce((sum, s) => sum + s.impactScore, 0) / scans.length
      : 0;

    // Retention rate (days with activity / total days)
    const uniqueDays = new Set(scans.map(s => s.scannedAt.toISOString().split('T')[0]));
    const retentionRate = (uniqueDays.size / days) * 100;

    // Top categories
    const categoryCount: Record<string, number> = {};
    allScans.forEach(scan => {
      categoryCount[scan.category] = (categoryCount[scan.category] || 0) + 1;
    });

    const topCategories = Object.entries(categoryCount)
      .map(([category, count]) => ({ category, count }))
      .sort((a, b) => b.count - a.count)
      .slice(0, 5);

    // Peak hours
    const hourCount: Record<number, number> = {};
    events.forEach(event => {
      const hour = event.timestamp.getHours();
      hourCount[hour] = (hourCount[hour] || 0) + 1;
    });

    const peakHours = Object.entries(hourCount)
      .map(([hour, count]) => ({ hour: parseInt(hour), count }))
      .sort((a, b) => b.count - a.count)
      .slice(0, 5);

    return {
      dailyScans,
      missionCompletion: Math.round(missionCompletion * 10) / 10,
      averageImpact: Math.round(averageImpact * 10) / 10,
      retentionRate: Math.round(retentionRate * 10) / 10,
      topCategories,
      peakHours
    };
  }

    totalUsers: number;
    activeUsers: number;
    totalScans: number;
    scansPerDay: number;
    retentionRate: number;
    subscriptionRate: number;
    topEvents: Array<{ eventName: string; count: number }>;
    platformBreakdown: Array<{ platform: string; count: number }>;
  }> {
    const startDate = new Date();
    startDate.setDate(startDate.getDate() - days);

    const [
      totalUsers,
      activeUsers,
      totalScans,
      scansToday,
      subscriptions,
      events,
      platformData
    ] = await Promise.all([
      prisma.user.count(),
      prisma.user.count({
        where: { lastLoginAt: { gte: startDate } }
      }),
      prisma.scan.count({ where: { scannedAt: { gte: startDate } } }),
      prisma.scan.count({
        where: {
          scannedAt: {
            gte: new Date(new Date().setHours(0, 0, 0, 0))
          }
        }
      }),
      prisma.subscription.count({ where: { tier: { not: 'FREE' } } }),
      prisma.analyticsEvent.groupBy({
        by: ['eventName'],
        where: { timestamp: { gte: startDate } },
        _count: { eventName: true },
        take: 10,
        orderBy: { _count: { eventName: 'desc' } }
      }),
      prisma.analyticsEvent.groupBy({
        by: ['platform'],
        where: {
          timestamp: { gte: startDate },
          platform: { not: null }
        },
        _count: { platform: true }
      })
    ]);

    const scansPerDay = totalScans / days;
    const retentionRate = activeUsers / totalUsers * 100;
    const subscriptionRate = subscriptions / totalUsers * 100;

    const topEvents = events.map(event => ({
      eventName: event.eventName,

    }));

    const platformBreakdown = platformData.map(data => ({

      count: data._count.platform
    }));

    return {

  async getRetentionReport(userId: string): Promise<{
    dailyStatus: {

      worstDrink: { name: string; score: number };

      bloodSugarData: number[];
      cleanDrinkStreak: number;
      insights: string[];

    reengagementTriggers: string[];

        orderBy: { scannedAt: 'desc' },
        take: 7,
        include: {

        }

    // Calculate daily status
    const recentScans = scans.slice(0, 3);
    const avgScore = recentScans.length > 0
      ? recentScans.reduce((sum, s) => sum + s.impactScore, 0) / recentScans.length
      : 75;

    const hydrationAvg = recentScans.length > 0
      ? recentScans.reduce((sum, s) => sum + s.hydrationLevel, 0) / recentScans.length
      : 68;

    const status = this.getStatusFromScore(avgScore);
    const trend = this.calculateTrend(scans.map(s => s.impactScore));
    const dehydrationRisk = hydrationAvg < 60;

    const dailyStatus = {
      score: Math.round(avgScore),
      status,
      trend: Math.round(trend * 10) / 10,
      hydration: Math.round(hydrationAvg),
      energyPrediction: this.getEnergyPrediction(avgScore),
      bloodSugarStability: this.getBloodSugarStability(avgScore),
      recommendation: this.getRecommendation(status, hydrationAvg),
      dehydrationRisk
    };

    // Calculate or use weekly summary
    let weeklySummary;
    if (weeklyReport) {
      weeklySummary = {
        averageImpact: weeklyReport.averageImpact,
        trend: weeklyReport.trend,
        bestDrink: { name: 'Green Detox Tea', score: 92 }, // Would come from DB
        worstDrink: { name: 'Caramel Frappé', score: 34 }, // Would come from DB
        hydrationTrend: weeklyReport.hydrationTrend,
        bloodSugarData: weeklyReport.bloodSugarData,
        cleanDrinkStreak: weeklyReport.cleanDrinkStreak,
        insights: weeklyReport.insights
      };
    } else {
      weeklySummary = {
        averageImpact: avgScore,
        trend: 0,
        bestDrink: { name: 'Green Tea', score: 85 },
        worstDrink: { name: 'Soda', score: 35 },
        hydrationTrend: [65, 72, 68, 75, 80, 78, 82],
        bloodSugarData: [72, 68, 75, 70, 65, 73, 71],
        cleanDrinkStreak: streaks?.cleanDrinksStreak || 0,

    const reengagementTriggers = this.generateTriggers(

      weeklySummary,
      reengagementTriggers
    };
  }

  private getStatusFromScore(score: number): 'Optimal' | 'Stable' | 'Risky' | 'Damaging' {
    if (score >= 80) return 'Optimal';
    if (score >= 60) return 'Stable';
    if (score >= 40) return 'Risky';
    return 'Damaging';
  }

  private calculateTrend(scores: number[]): number {
    if (scores.length < 2) return 0;

    const recent = scores.slice(0, 3).reduce((a, b) => a + b, 0) / Math.min(3, scores.length);
    const previous = scores.slice(3, 6).reduce((a, b) => a + b, 0) / Math.min(3, scores.length);

    if (previous === 0) return 0;
    return ((recent - previous) / previous) * 100;
  }

  private getEnergyPrediction(score: number): string {
    if (score >= 80) return 'High';
    if (score >= 60) return 'Medium';
    return 'Low';
  }

  private getBloodSugarStability(score: number): string {
    if (score >= 80) return 'Very Stable';
    if (score >= 60) return 'Stable';
    if (score >= 40) return 'Moderate';
    return 'Unstable';
  }

  private getRecommendation(status: string, hydration: number): string {
    if (status === 'Damaging') {
      return 'Consider switching to water or unsweetened drinks for your next beverage.';
    } else if (hydration < 60) {
      return 'Start with 600ml water to stabilize hydration levels.';
    } else if (status === 'Optimal') {
      return 'Great choices! Keep prioritizing natural, low-sugar drinks.';
    }
    return 'Try adding more water between other drinks today.';
  }

  private generateTriggers(
    dailyStatus: any,
    streaks: any,
    recentMissions: number
  ): string[] {
    const triggers: string[] = [];

    if (dailyStatus.hydration < 60) {
      triggers.push('Your hydration may drop later today. Drink water now!');
    }

    if (dailyStatus.energyPrediction === 'Low') {
      triggers.push('Check back in 2 hours to see your energy levels');
    }

    if (streaks?.hydrationStreak >= 5) {
      triggers.push(`You're on a ${streaks.hydrationStreak}-day hydration streak! Keep it up!`);
    }

    if (recentMissions === 0) {
      triggers.push('Complete a mission today to boost your impact score!');
    }

    if (dailyStatus.score < 50) {
      triggers.push('Your recent drinks are impacting your score. Try healthier options!');
    }

    return triggers.slice(0, 3); // Return max 3 triggers
  }

        // Try cache first
        const cached = await redis.get(cacheKey);
        if (cached) {
            return JSON.parse(cached);
        }

        // Cache miss - compute from database
        const stats = await computeUserStats(userId);

        // Store in cache
        await redis.setex(
            cacheKey,
            this.TTL.USER_STATS,
            JSON.stringify(stats)
        );

        return stats;
    }

    async cacheScanResult(scanId: string, result: any): Promise<void> {
        const cacheKey = `scan_result:${scanId}`;

        await redis.setex(
            cacheKey,
            this.TTL.SCAN_RESULTS,
            JSON.stringify(result)
        );
    }

    async invalidateUserCache(userId: string): Promise<void> {
        const patterns = [
            `user_stats:${userId}`,
            `user_missions:${userId}`,
            `user_streak:${userId}`
        ];

        for (const pattern of patterns) {
            const keys = await redis.keys(pattern);
            if (keys.length > 0) {
                await redis.del(...keys);
            }
        }
    }

  async createMission(userId: string, data: MissionInput): Promise<Mission> {
    const mission = await prisma.mission.create({
      data: {
        id: uuidv4(),
        userId,
        title: data.title,
        description: data.description,
        type: data.type as MissionType,
        target: data.target,
        progress: data.progress || 0,
        rewardScore: data.rewardScore || 0,
        rewardImpact: data.rewardImpact || 0,
        rewardText: data.rewardText,
        expiresAt: new Date(data.expiresAt)
      }
    });

    return mission;
  }

  async getMissions(userId: string, options?: {
    completed?: boolean;
    expiresAfter?: Date;
    expiresBefore?: Date;
    type?: MissionType;
  }): Promise<Mission[]> {
    const where: any = { userId };

    if (options?.completed !== undefined) {
      where.completed = options.completed;
    }

    if (options?.expiresAfter || options?.expiresBefore) {
      where.expiresAt = {};
      if (options.expiresAfter) where.expiresAt.gte = options.expiresAfter;
      if (options.expiresBefore) where.expiresAt.lte = options.expiresBefore;
    }

    if (options?.type) {
      where.type = options.type;
    }

    return await prisma.mission.findMany({
      where,
      orderBy: [
        { completed: 'asc' },
        { expiresAt: 'asc' },
        { createdAt: 'desc' }
      ]
    });
  }

  async completeMission(userId: string, missionId: string, progressDelta?: number): Promise<Mission> {
    return await prisma.$transaction(async (tx) => {
      const mission = await tx.mission.findFirst({
        where: {
          id: missionId,
          userId,
          completed: false
        }
      });

      if (!mission) {
        throw new Error('Mission not found or already completed');
      }

      let newProgress = mission.progress;
      let isCompleted = mission.completed;

      if (progressDelta !== undefined) {
        newProgress = Math.min(mission.target, mission.progress + progressDelta);
        isCompleted = newProgress >= mission.target;
      } else {
        newProgress = mission.target;
        isCompleted = true;
      }

      const updatedMission = await tx.mission.update({
        where: { id: missionId },
        data: {
          progress: newProgress,
          completed: isCompleted,
          completedAt: isCompleted ? new Date() : null
        }
      });

      if (isCompleted && mission.rewardImpact > 0) {
        // Update user streaks based on mission type
        await this.updateStreaksForMission(userId, mission.type);

        // Log analytics
        await tx.analyticsEvent.create({
          data: {
            id: uuidv4(),
            userId,
            eventName: 'mission_completed',
            eventType: 'MISSION_COMPLETE',
            properties: {
              missionId,
              missionType: mission.type,
              rewardImpact: mission.rewardImpact
            }
          }
        });
      }

      return updatedMission;
    });
  }

  async updateStreaksForMission(userId: string, missionType: MissionType): Promise<void> {
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const streaks = await prisma.streak.findUnique({
      where: { userId }
    });

    if (!streaks) return;

    const updates: any = {};

    switch (missionType) {
      case MissionType.HYDRATION:
        updates.lastHydrationDate = new Date();
        updates.hydrationStreak = { increment: 1 };
        if (streaks.hydrationStreak + 1 > streaks.bestHydrationStreak) {
          updates.bestHydrationStreak = streaks.hydrationStreak + 1;
        }
        break;

      case MissionType.CLEAN_DRINK:
      case MissionType.SUGAR:
        updates.lastCleanDrinkDate = new Date();
        updates.cleanDrinksStreak = { increment: 1 };
        if (streaks.cleanDrinksStreak + 1 > streaks.bestCleanStreak) {
          updates.bestCleanStreak = streaks.cleanDrinksStreak + 1;
        }
        break;
    }

    await prisma.streak.update({
      where: { userId },
      data: updates
    });
  }

  async getMissionStats(userId: string): Promise<{
    totalCompleted: number;
    totalImpact: number;
    currentStreak: number;
    bestStreak: number;
    todayCompleted: number;
    todayTotal: number;
  }> {
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const tomorrow = new Date(today);
    tomorrow.setDate(tomorrow.getDate() + 1);

    const [allMissions, todayMissions, streaks] = await Promise.all([
      prisma.mission.findMany({
        where: { userId, completed: true },
        select: { rewardImpact: true }
      }),
      prisma.mission.findMany({
        where: {
          userId,
          expiresAt: { gte: today, lt: tomorrow }
        }
      }),
      prisma.streak.findUnique({ where: { userId } })
    ]);

    const todayCompleted = todayMissions.filter(m => m.completed).length;

    return {
      totalCompleted: allMissions.length,
      totalImpact: allMissions.reduce((sum, m) => sum + (m.rewardImpact || 0), 0),
      currentStreak: streaks?.scanningStreak || 0,
      bestStreak: streaks?.bestOverallStreak || 0,
      todayCompleted,
      todayTotal: todayMissions.length
    };
  }

  async getUserScans(userId: string, options?: {
    limit?: number;
    offset?: number;
    startDate?: Date;
    endDate?: Date;
    category?: string;
    minScore?: number;
    maxScore?: number;
  }): Promise<{ scans: Scan[], total: number }> {
    const where: any = { userId };

    if (options?.startDate || options?.endDate) {
      where.scannedAt = {};
      if (options.startDate) where.scannedAt.gte = options.startDate;
      if (options.endDate) where.scannedAt.lte = options.endDate;
    }

    if (options?.category) {
      where.category = options.category;
    }

    if (options?.minScore || options?.maxScore) {
      where.impactScore = {};
      if (options.minScore) where.impactScore.gte = options.minScore;
      if (options.maxScore) where.impactScore.lte = options.maxScore;
    }

    const [scans, total] = await Promise.all([
      prisma.scan.findMany({
        where,
        include: {
          composition: {
            include: {
              ingredients: true
            }
          },
          report: true,
          aiInsight: true
        },
        orderBy: { scannedAt: 'desc' },
        take: options?.limit || 20,
        skip: options?.offset || 0
      }),
      prisma.scan.count({ where })
    ]);

    return { scans, total };
  }

  async getUserStats(userId: string): Promise<{
    totalScans: number;
    averageScore: number;
    bestScore: number;
    worstScore: number;
    categoryBreakdown: Record<string, number>;
    weeklyAverage: number;
    streak: number;
  }> {
    const [totalScans, scans, streaks] = await Promise.all([
      prisma.scan.count({ where: { userId } }),
      prisma.scan.findMany({
        where: { userId },
        select: {
          impactScore: true,
          category: true,
          scannedAt: true
        },
        orderBy: { scannedAt: 'desc' },
        take: 100
      }),
      prisma.streak.findUnique({ where: { userId } })
    ]);

    if (scans.length === 0) {
      return {
        totalScans: 0,
        averageScore: 0,
        bestScore: 0,
        worstScore: 0,
        categoryBreakdown: {},
        weeklyAverage: 0,
        streak: streaks?.scanningStreak || 0
      };
    }

    const scores = scans.map(s => s.impactScore);
    const averageScore = scores.reduce((a, b) => a + b, 0) / scores.length;
    const bestScore = Math.max(...scores);
    const worstScore = Math.min(...scores);

    // Calculate category breakdown
    const categoryBreakdown: Record<string, number> = {};
    scans.forEach(scan => {
      categoryBreakdown[scan.category] = (categoryBreakdown[scan.category] || 0) + 1;
    });

    // Calculate weekly average (last 7 days)
    const oneWeekAgo = new Date();
    oneWeekAgo.setDate(oneWeekAgo.getDate() - 7);
    const weeklyScans = scans.filter(s => s.scannedAt >= oneWeekAgo);
    const weeklyAverage = weeklyScans.length > 0
      ? weeklyScans.reduce((sum, s) => sum + s.impactScore, 0) / weeklyScans.length
      : averageScore;

    return {
      totalScans,
      averageScore,
      bestScore,
      worstScore,
      categoryBreakdown,
      weeklyAverage,
      streak: streaks?.scanningStreak || 0
    };
  }

  async getWeeklyTrends(userId: string, weeks: number = 4): Promise<any[]> {
    const trends = [];
    const now = new Date();

    for (let i = weeks - 1; i >= 0; i--) {
      const weekStart = new Date(now);
      weekStart.setDate(weekStart.getDate() - (i * 7));
      const weekEnd = new Date(weekStart);
      weekEnd.setDate(weekEnd.getDate() + 6);

      const weekScans = await prisma.scan.findMany({
        where: {
          userId,
          scannedAt: {
            gte: weekStart,
            lte: weekEnd
          }
        }
      });

      const scores = weekScans.map(s => s.impactScore);
      const averageScore = scores.length > 0
        ? scores.reduce((a, b) => a + b, 0) / scores.length
        : 0;

      const hydrationScores = weekScans.map(s => s.hydrationLevel);
      const averageHydration = hydrationScores.length > 0
        ? hydrationScores.reduce((a, b) => a + b, 0) / hydrationScores.length
        : 0;

      trends.push({
        date: weekStart.toISOString().split('T')[0],
        averageScore: Math.round(averageScore * 10) / 10,
        totalDrinks: weekScans.length,
        hydrationScore: Math.round(averageHydration * 10) / 10
      });
    }

    return trends;
  }

  private async updateStreaks(userId: string): Promise<void> {
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const streaks = await prisma.streak.findUnique({
      where: { userId }
    });

    if (!streaks) return;

    const lastScan = streaks.lastScanDate ? new Date(streaks.lastScanDate) : null;
    const isNewDay = !lastScan || lastScan < today;

    const updates: any = {
      lastScanDate: new Date()
    };

    if (isNewDay) {
      // Check if yesterday was scanned
      const yesterday = new Date(today);
      yesterday.setDate(yesterday.getDate() - 1);

      if (lastScan && lastScan >= yesterday) {
        // Continue streak
        updates.scanningStreak = { increment: 1 };
        if (streaks.scanningStreak + 1 > streaks.bestOverallStreak) {
          updates.bestOverallStreak = streaks.scanningStreak + 1;
        }
      } else {
        // Reset streak
        updates.scanningStreak = 1;
      }
    }

    await prisma.streak.update({
      where: { userId },
      data: updates
    });
  }

  ): Promise<string> {

  async handleStripeWebhook(event: Stripe.Event): Promise<void> {
    switch (event.type) {
      case 'customer.subscription.created':
      case 'customer.subscription.updated':
        await this.handleSubscriptionUpdate(event.data.object as Stripe.Subscription);
        break;

      case 'customer.subscription.deleted':
        await this.handleSubscriptionCancel(event.data.object as Stripe.Subscription);
        break;

      case 'checkout.session.completed':
        await this.handleCheckoutComplete(event.data.object as Stripe.Checkout.Session);
        break;
    }
  }

  private async handleSubscriptionUpdate(subscription: Stripe.Subscription): Promise<void> {
    const customerId = subscription.customer as string;

    const dbSubscription = await prisma.subscription.findFirst({
      where: { stripeCustomerId: customerId }
    });

    if (dbSubscription) {
      await prisma.subscription.update({
        where: { id: dbSubscription.id },
        data: {
          stripeSubscriptionId: subscription.id,
          status: subscription.status.toUpperCase() as SubscriptionStatus,
          currentPeriodStart: new Date(subscription.current_period_start * 1000),
          currentPeriodEnd: new Date(subscription.current_period_end * 1000)
        }
      });
    }
  }

  private async handleSubscriptionCancel(subscription: Stripe.Subscription): Promise<void> {
    const dbSubscription = await prisma.subscription.findFirst({
      where: { stripeSubscriptionId: subscription.id }
    });

    if (dbSubscription) {
      await prisma.subscription.update({
        where: { id: dbSubscription.id },
        data: {
          status: 'CANCELED',
          canceledAt: new Date()
        }
      });
    }
  }

  private async handleCheckoutComplete(session: Stripe.Checkout.Session): Promise<void> {
    const { userId, tier, billingCycle } = session.metadata || {};

    if (userId && tier && billingCycle) {
      await this.createSubscription(userId, { tier, billingCycle } as CreateSubscriptionInput);
    }

      where: { id: userId },
      select: { email: true, displayName: true }
    });

    if (!user) {
      throw new Error('User not found');
    }

    return await this.createStripeCustomer(userId, user.email, user.displayName || undefined);

    });
  }

    return await prisma.user.update({
      where: { id: userId },

        avatarUrl: data.avatarUrl,
        username: data.username
      }
    });
  }

  async updatePreferences(userId: string, data: UpdatePreferencesInput): Promise<UserPreferences> {
    return await prisma.userPreferences.update({
      where: { userId },
      data
    });
  }

  async updateLastLogin(userId: string): Promise<void> {
    await prisma.user.update({
      where: { id: userId },
      data: { lastLoginAt: new Date() }
    });
  }

  async verifyPassword(user: User, password: string): Promise<boolean> {
    if (!user.password) return false;
    return await bcrypt.compare(password, user.password);
  }

  async changePassword(userId: string, currentPassword: string, newPassword: string): Promise<void> {
    const user = await prisma.user.findUnique({
      where: { id: userId },
      select: { password: true }
    });

    if (!user || !user.password) {
      throw new Error('User not found');
    }

    const isValid = await bcrypt.compare(currentPassword, user.password);
    if (!isValid) {
      throw new Error('Current password is incorrect');
    }

    const hashedPassword = await bcrypt.hash(newPassword, 10);

    await prisma.user.update({
      where: { id: userId },
      data: { password: hashedPassword }
    });
  }

  async deleteAccount(userId: string): Promise<void> {
    await prisma.user.update({
      where: { id: userId },
      data: { deletedAt: new Date() }
    });
  }

  async getUserStats(userId: string): Promise<{
    totalScans: number;
    streak: number;
    averageImpact: number;
    hydrationScore: number;
    missionCompletion: number;
    subscriptionTier: string;
  }> {
    const [user, scans, missions, todayMissions] = await Promise.all([
      this.getUserById(userId),
      prisma.scan.findMany({
        where: { userId },
        select: { impactScore: true, hydrationLevel: true }
      }),
      prisma.mission.findMany({
        where: { userId, completed: true }
      }),
      prisma.mission.findMany({
        where: {
          userId,
          expiresAt: { gte: new Date() },
          createdAt: { gte: new Date(new Date().setHours(0, 0, 0, 0)) }
        }
      })
    ]);

    const totalScans = scans.length;
    const averageImpact = scans.length > 0
      ? scans.reduce((sum, s) => sum + s.impactScore, 0) / scans.length
      : 0;
    const hydrationScore = scans.length > 0
      ? scans.reduce((sum, s) => sum + s.hydrationLevel, 0) / scans.length
      : 0;
    const missionCompletion = todayMissions.length > 0
      ? (todayMissions.filter(m => m.completed).length / todayMissions.length) * 100
      : 0;

    return {
      totalScans,
      streak: user?.streak.scanningStreak || 0,
      averageImpact: Math.round(averageImpact * 10) / 10,
      hydrationScore: Math.round(hydrationScore * 10) / 10,
      missionCompletion: Math.round(missionCompletion),
      subscriptionTier: user?.subscription?.tier || 'FREE'
    };
  }

  async searchUsers(query: string, limit: number = 10): Promise<User[]> {

      where: {
        OR: [

        username: true,
        avatarUrl: true,
        streak: {
          select: {
            scanningStreak: true,

          }
        }
      }
    });
  }

export class InsightEngine {
  // 1. USER STAGE SYSTEM (progression layer)
  async getUserStage(userId: string) {
    const user = await prisma.user.findUnique({
      where: { id: userId },
      include: { streak: true }
    });

    const days = (user as any)?.streak?.scanningStreak || 0;

    if (days <= 7) return 'week1';
    if (days <= 21) return 'week2';
    return 'veteran';
  }

  // 2. CONTEXT AWARE INSIGHT (KEY UPGRADE)
  async generateInsight(userId: string, scan: any) {
    const stage = await this.getUserStage(userId);

    const recentScans = await prisma.scan.findMany({
      where: { userId },
      orderBy: { createdAt: 'desc' },
      take: 10,
    });

    const sodaCount = recentScans.filter((s: any) => s.category === 'soda').length;

    let message = "";

    // 🧠 Progressive insight logic
    if (stage === 'week1') {
      message = scan.impactScore > 70
        ? "High impact drink detected."
        : "Basic nutritional breakdown ready.";
    } else if (stage === 'week2') {
      message = sodaCount >= 2
        ? "You are repeating a pattern with sugary drinks."
        : "Your choices are becoming more aware.";
    } else if (stage === 'veteran') {
      message = sodaCount >= 3
        ? "Your pattern suggests dependency loop forming."
        : "Your metabolic behavior is stabilizing.";
    }

    return {
      stage,
      message,
      pattern: sodaCount,
      scanId: scan.id,
      cta: sodaCount > 2 ? "Try a low-sugar swap" : null
    };
  }
}

export class StreakService {
  async update(userId: string): Promise<any> {
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const streaks = await prisma.streak.findUnique({
      where: { userId }
    });

    if (!streaks) {
      return await prisma.streak.create({
        data: {
          userId,
          lastScanDate: new Date(),
          scanningStreak: 1,
          bestOverallStreak: 1
        }
      });
    }

    const lastScan = streaks.lastScanDate ? new Date(streaks.lastScanDate) : null;
    const isNewDay = !lastScan || lastScan < today;

    if (!isNewDay) return streaks;

    const yesterday = new Date(today);
    yesterday.setDate(yesterday.getDate() - 1);

    const updates: any = {
      lastScanDate: new Date()
    };

    if (lastScan && lastScan >= yesterday) {
      // Continue streak
      updates.scanningStreak = { increment: 1 };
      // Check for best overall streak will be done after update or via middleware/trigger if possible,
      // but here we just increment.
    } else {
      // Reset streak
      updates.scanningStreak = 1;
    }

    return await prisma.streak.update({
      where: { userId },
      data: updates
    });
  }
}
