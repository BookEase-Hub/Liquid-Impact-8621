```typescript
// src/hooks/useApp.ts
import { useState, useCallback } from 'react';
import AsyncStorage from '@react-native-async-storage/async-storage';

interface User {
  id: string;
  email: string;
  displayName?: string;
  avatarUrl?: string;
}

interface AppState {
  user: User | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  errorMessage: string | null;
  subscriptionTier: 'free' | 'starter' | 'pro' | 'elite';
  dailyScanCount: number;
  streakDays: number;
}

const initialState: AppState = {
  user: null,
  isAuthenticated: false,
  isLoading: false,
  errorMessage: null,
  subscriptionTier: 'free',
  dailyScanCount: 0,
  streakDays: 0,
};

export function useApp() {
  const [state, setState] = useState<AppState>(initialState);

  const authenticate = useCallback(async (email: string, password: string) => {
    setState(prev => ({ ...prev, isLoading: true, errorMessage: null }));
    try {
      // TODO: Replace with actual API call
      await new Promise(resolve => setTimeout(resolve, 1000));
      
      const mockUser: User = {
        id: 'user_123',
        email,
        displayName: 'Demo User',
      };
      
      await AsyncStorage.setItem('@user', JSON.stringify(mockUser));
      await AsyncStorage.setItem('@auth_token', 'mock_token');
      
      setState(prev => ({
        ...prev,
        user: mockUser,
        isAuthenticated: true,
        isLoading: false,
      }));
    } catch (error: any) {
      setState(prev => ({
        ...prev,
        errorMessage: error.message || 'Authentication failed',
        isLoading: false,
      }));
      throw error;
    }
  }, []);

  const register = useCallback(async (email: string, password: string, displayName: string) => {
    setState(prev => ({ ...prev, isLoading: true, errorMessage: null }));
    try {
      // TODO: Replace with actual API call
      await new Promise(resolve => setTimeout(resolve, 1000));
      
      const mockUser: User = {
        id: 'user_123',
        email,
        displayName,
      };
      
      await AsyncStorage.setItem('@user', JSON.stringify(mockUser));
      await AsyncStorage.setItem('@auth_token', 'mock_token');
      
      setState(prev => ({
        ...prev,
        user: mockUser,
        isAuthenticated: true,
        isLoading: false,
      }));
    } catch (error: any) {
      setState(prev => ({
        ...prev,
        errorMessage: error.message || 'Registration failed',
        isLoading: false,
      }));
      throw error;
    }
  }, []);

  const logout = useCallback(async () => {
    try {
      await AsyncStorage.multiRemove(['@user', '@auth_token']);
      setState(initialState);
    } catch (error) {
      console.error('Logout error:', error);
    }
  }, []);

  const updateUser = useCallback(async (updates: Partial<User>) => {
    if (!state.user) return;
    
    const updatedUser = { ...state.user, ...updates };
    await AsyncStorage.setItem('@user', JSON.stringify(updatedUser));
    setState(prev => ({ ...prev, user: updatedUser }));
  }, [state.user]);

  const loadUser = useCallback(async () => {
    try {
      const userJson = await AsyncStorage.getItem('@user');
      if (userJson) {
        const user: User = JSON.parse(userJson);
        setState(prev => ({
          ...prev,
          user,
          isAuthenticated: true,
        }));
      }
    } catch (error) {
      console.error('Load user error:', error);
    }
  }, []);

  return {
    ...state,
    authenticate,
    register,
    logout,
    updateUser,
    loadUser,
  };
}
```

```typescript
// src/features/settings/components/AmbientBackground.tsx
import React, { useRef, useEffect } from 'react';
import { View, StyleSheet, Animated } from 'react-native';
import { THEME } from '../theme/colors';

export const AmbientBackground: React.FC = () => {
  const orb1Anim = useRef(new Animated.Value(0)).current;
  const orb2Anim = useRef(new Animated.Value(0)).current;

  useEffect(() => {
    // Gentle floating animation
    const animateOrb = (anim: Animated.Value, duration: number, offset: number) => {
      Animated.loop(
        Animated.sequence([
          Animated.timing(anim, { toValue: 1, duration, delay: offset, useNativeDriver: true }),
          Animated.timing(anim, { toValue: 0, duration, delay: offset, useNativeDriver: true }),
        ])
      ).start();
    };
    
    animateOrb(orb1Anim, 4000, 0);
    animateOrb(orb2Anim, 5000, 1000);
    
    return () => { 
      orb1Anim.stopAnimation(); 
      orb2Anim.stopAnimation(); 
    };
  }, [orb1Anim, orb2Anim]);

  return (
    <View style={styles.container} pointerEvents="none">
      <Animated.View style={[
        styles.orb,
        {
          backgroundColor: THEME.glowCyan,
          transform: [
            { translateX: orb1Anim.interpolate({ inputRange: [0, 1], outputRange: [-30, 30] }) },
            { translateY: orb1Anim.interpolate({ inputRange: [0, 1], outputRange: [-20, 20] }) },
          ],
        },
      ]} />
      <Animated.View style={[
        styles.orb,
        {
          backgroundColor: THEME.glowPurple,
          right: 20,
          bottom: 100,
          width: 180,
          height: 180,
          transform: [
            { translateX: orb2Anim.interpolate({ inputRange: [0, 1], outputRange: [20, -20] }) },
            { translateY: orb2Anim.interpolate({ inputRange: [0, 1], outputRange: [30, -30] }) },
          ],
        },
      ]} />
    </View>
  );
};
```

```typescript
  accentPurple: '#7B2CBF',

  glassBg: 'rgba(255,255,255,0.06)',
```

```typescript
};

export default function ProfileScreen() {
  const router = useRouter();
  const insets = useSafeAreaInsets();
  const { user, updateUser } = useApp();
  
  const [displayName, setDisplayName] = useState(user?.displayName || '');
  const [email, setEmail] = useState(user?.email || '');
  const [avatarUrl, setAvatarUrl] = useState(user?.avatarUrl || '');
  const [isLoading, setIsLoading] = useState(false);
  const [isUploading, setIsUploading] = useState(false);

  useEffect(() => {
    if (user) {
      setDisplayName(user.displayName || '');
      setEmail(user.email || '');
      setAvatarUrl(user.avatarUrl || '');
    }
  }, [user]);

  const handlePickImage = async () => {
    const permissionResult = await ImagePicker.requestMediaLibraryPermissionsAsync();
    
    if (permissionResult.granted === false) {
      Alert.alert('Permission Required', 'Permission to access camera roll is required!');
      return;
    }

    const result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsEditing: true,
      aspect: [1, 1],
      quality: 0.8,
    });

    if (!result.canceled && result.assets[0]) {
      setIsUploading(true);
      await Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium);
      
      // In production, upload to S3/Cloudinary here
      // For now, use local URI
      setAvatarUrl(result.assets[0].uri);
      setIsUploading(false);
    }
  };

    if (!displayName.trim()) {
      Alert.alert('Error', 'Display name is required');
      return;
    }

    setIsLoading(true);
    await Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium);

    try {
      // In production, call API to update profile
      // await api.updateProfile({ displayName, avatarUrl });
      
      // For now, update local state
      await updateUser({ displayName, avatarUrl });
      
      // Persist to AsyncStorage
      await AsyncStorage.setItem('@user_profile', JSON.stringify({
        displayName,
        avatarUrl,
      }));

      Alert.alert('Success', 'Profile updated successfully');
      router.back();
    } catch (error) {
      Alert.alert('Error', 'Failed to update profile');
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <View style={[styles.container, { paddingTop: insets.top }]}>
      <StatusBar barStyle="light-content" backgroundColor={THEME.background} />

      <View style={styles.header}>

          style={styles.backButton}

        </TouchableOpacity>

          onPress={handleSave}

          ) : (
            <Text style={styles.saveText}>Save</Text>

        {/* Avatar Section */}

            <View style={styles.avatarContainer}>
```

```typescript
      Alert.alert('Error', passwordError);
      return;

    if (newPassword !== confirmPassword) {
      Alert.alert('Error', 'New passwords do not match');
      return;

      return;
```

const WINDOW_MS = 60 * 1000; // 1 minute
const MAX_REQUESTS = 100;

// In‑memory store (replace with Redis for production)
const hits = new Map<string, { count: number; resetTime: number }>();

export const rateLimiter = async (req: Request, _res: Response, next: NextFunction) => {
  const key = req.ip || 'unknown';
  const now = Date.now();

  let record = hits.get(key);
  if (!record || now > record.resetTime) {
    record = { count: 1, resetTime: now + WINDOW_MS };
    hits.set(key, record);
  } else {
    record.count++;
  }

  // Log to database asynchronously
  await prisma.rateLimitLog.create({
    data: {
      endpoint: req.path,
      ip: req.ip || '',
      userId: (req as any).user?.id,
      timestamp: new Date(),
      action: 'api_request',
      blocked: record.count > MAX_REQUESTS,
    },
  }).catch(() => {}); // silent failure

  if (record.count > MAX_REQUESTS) {
    return next(new Error('Too many requests'));
  }
  next();
};

export class AuthService {
  static async signUp(email: string, password: string, displayName?: string) {
    const existing = await prisma.user.findUnique({ where: { email } });
    if (existing) throw new AppError('Email already exists', 400);

    const hashedPassword = await bcrypt.hash(password, 12);
    const user = await prisma.user.create({
      data: { email, password: hashedPassword, displayName },
    });

    const token = this.generateToken(user.id);
    return { user, token };
  }

  static async signIn(email: string, password: string) {
    const user = await prisma.user.findUnique({ where: { email } });
    if (!user || !user.password) throw new AppError('Invalid credentials', 401);

    const valid = await bcrypt.compare(password, user.password);
    if (!valid) throw new AppError('Invalid credentials', 401);

    const token = this.generateToken(user.id);
    return { user, token };
  }

  private static generateToken(userId: string): string {
    return jwt.sign({ userId }, env.JWT_SECRET, { expiresIn: '30d' });
  }
}

export class RevenueCatService {
  static async processEvent(event: any) {
    // Store webhook event for audit
    await prisma.webhookEvent.create({
      data: {
        source: 'revenuecat',
        eventType: event.type,
        appUserId: event.app_user_id,
        productId: event.product_id,
        payload: event,
      },
    });

    // Update subscription based on event type
    if (['INITIAL_PURCHASE', 'RENEWAL', 'PRODUCT_CHANGE'].includes(event.type)) {
      const tier = event.product_id.split('_')[0].toUpperCase() as any;
      const billingCycle = event.product_id.includes('yearly') ? 'YEARLY' : 'MONTHLY';
      const platform = event.store === 'app_store' ? 'APPLE' : 'GOOGLE';

      await prisma.subscription.upsert({
        where: { userId: event.app_user_id },
        update: {
          tier,
          status: 'ACTIVE',
          billingCycle,
          platform,
          revenueCatCustomerId: event.app_user_id,
          storeProductId: event.product_id,
          storeTransactionId: event.transaction_id,
          currentPeriodEnd: new Date(event.expiration_at_millis),
        },
        create: {
          userId: event.app_user_id,
          tier,
          status: 'ACTIVE',
          billingCycle,
          platform,
          revenueCatCustomerId: event.app_user_id,
          storeProductId: event.product_id,
          storeTransactionId: event.transaction_id,
          currentPeriodEnd: new Date(event.expiration_at_millis),
        },
      });
    } else if (event.type === 'CANCELLATION') {
      await prisma.subscription.update({
        where: { userId: event.app_user_id },
        data: { status: 'CANCELED', canceledAt: new Date() },
      });
    }
  }

  static async restore(userId: string, receipt: string) {
    // Call RevenueCat REST API to verify receipt
    const response = await axios.post('https://api.revenuecat.com/v1/receipts', {
      receipt,
      app_user_id: userId,
      fetch_products: false,
    }, {
      headers: { Authorization: `Bearer ${env.REVENUECAT_SECRET}` },
    });
    // Update subscription based on response
    return response.data;
  }
}

export async function processScanJob(job: any) {
  const scan = await prisma.scan.findUnique({
    where: { id: job.payload.scanId },
    include: { composition: true },
  });
  if (!scan) throw new Error('Scan not found');

  // 1. Call OpenAI Vision to analyze image
  const visionResult = await AiService.analyzeImage(scan.imageUrl);

  // 2. Create/update composition
  const composition = await prisma.composition.create({
    data: {
      calories: visionResult.calories,
      sugarGrams: visionResult.sugarGrams,
      caffeineMg: visionResult.caffeineMg,
      sodiumMg: visionResult.sodiumMg,
      additives: visionResult.additives,
      artificialSweeteners: visionResult.artificialSweeteners,
      scanId: scan.id,
    },
  });

  // 3. Create ingredients
  for (const ing of visionResult.ingredients) {
    await prisma.ingredient.create({
      data: {
        name: ing.name,
        function: ing.function,
        healthRole: ing.healthRole,
        riskLevel: ing.riskLevel,
        position: ing.position,
        compositionId: composition.id,
      },
    });
  }

  // 4. Generate report using GPT-4
  const report = await AiService.generateReport(composition);
  await prisma.report.create({ data: { ...report, scanId: scan.id } });

  // 5. Update scan status
  await prisma.scan.update({
    where: { id: scan.id },
    data: { status: 'OPTIMAL', compositionId: composition.id },
  });

  // 6. Enqueue analytics update
  await JobService.enqueue(scan.userId, 'ANALYTICS' as any);
}
