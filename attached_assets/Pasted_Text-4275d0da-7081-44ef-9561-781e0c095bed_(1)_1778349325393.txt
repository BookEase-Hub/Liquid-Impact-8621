### `ColorExtensions.swift`
```swift
import SwiftUI

extension Color {
    init(hex: String) {
        let hex = hex.trimmingCharacters(in: CharacterSet.alphanumerics.inverted)
        var int: UInt64 = 0
        Scanner(string: hex).scanHexInt64(&int)
        let a, r, g, b: UInt64
        switch hex.count {
        case 3: (a, r, g, b) = (255, (int >> 8) * 17, (int >> 4 & 0xF) * 17, (int & 0xF) * 17)
        case 6: (a, r, g, b) = (255, int >> 16, int >> 8 & 0xFF, int & 0xFF)
        case 8: (a, r, g, b) = (int >> 24, int >> 16 & 0xFF, int >> 8 & 0xFF, int & 0xFF)
        default: (a, r, g, b) = (255, 0, 0, 0)
        }
        self.init(
            .sRGB,
            red: Double(r) / 255,
            green: Double(g) / 255,
            blue: Double(b) / 255,
            opacity: Double(a) / 255
        )
    }
}
```

```swift
import SwiftUI

@MainActor
class NavigationCoordinator: ObservableObject {
    static let shared = NavigationCoordinator()
    
    @Published var selectedTab: AppTab = .home
    @Published var showScanSheet = false
    @Published var showPaywall = false
    @Published var showReport = false
    @Published var showSettings = false
    @Published var showOnboarding = false

    @Published var showNotificationPermission = false

        case home = "Home"
        case status = "Status"
        case scan = "Scan"
        case history = "History"
        case profile = "Profile"
        
        var id: String { rawValue }
        var iconName: String {
            switch self {
            case .home: return "house.fill"
            case .status: return "waveform.path.ecg"
            case .scan: return "camera.fill"
            case .history: return "clock.arrow.circlepath"
            case .profile: return "person.fill"
            }
        }
        var iconUnselected: String {
            switch self {
            case .home: return "house"
            case .status: return "waveform.path.ecg"
            case .scan: return "camera"
            case .history: return "clock.arrow.circlepath"
            case .profile: return "person"
            }
        }
    }
    
    func navigateToTab(_ tab: AppTab) {
        withAnimation(.spring(response: 0.3, dampingFraction: 0.7)) {
            selectedTab = tab
        }
        HapticManager.shared.trigger(.light)
    }
    
    func openScanSheet() {
        withAnimation(.spring(response: 0.3)) {
            showScanSheet = true
        }
        HapticManager.shared.trigger(.medium)
    }
    
    func openPaywall(reason: String? = nil) {
        withAnimation(.spring(response: 0.3)) {
            showPaywall = true
        }
        HapticManager.shared.trigger(.light)
    }
    
    func openReport(for scanId: UUID) {
        selectedScanId = scanId
        withAnimation(.spring(response: 0.3)) {
            showReport = true
        }
    }
    
    func resetScanFlow() {
        showScanSheet = false
        showReport = false
        selectedScanId = nil
    }
}
```

### `HapticManager.swift`
```swift
import SwiftUI

class HapticManager {
    static let shared = HapticManager()
    private init() {}
    
    enum HapticType {
        case light, medium, heavy
        case success, warning, error
        case selection
        
        func trigger() {
            switch self {
            case .light: UIImpactFeedbackGenerator(style: .light).impactOccurred()
            case .medium: UIImpactFeedbackGenerator(style: .medium).impactOccurred()
            case .heavy: UIImpactFeedbackGenerator(style: .heavy).impactOccurred()
            case .success: UINotificationFeedbackGenerator().notificationOccurred(.success)
            case .warning: UINotificationFeedbackGenerator().notificationOccurred(.warning)
            case .error: UINotificationFeedbackGenerator().notificationOccurred(.error)
            case .selection: UISelectionFeedbackGenerator().selectionChanged()
            }
        }
    }
    
    func trigger(_ type: HapticType) {
        #if targetEnvironment(simulator)
        // Silent on simulator
        #else
        type.trigger()
        #endif
    }
}
```

```swift
import Foundation

@MainActor
class APIService {
    static let shared = APIService()
    
    private let baseURL = "https://api.liquidimpact.app"
    private var accessToken: String?
    private var refreshToken: String?
    
    private var authHeader: [String: String] {
        guard let token = accessToken else { return [:] }
        return ["Authorization": "Bearer \(token)"]
    }
    
    private var session: URLSession {
        let config = URLSessionConfiguration.default
        config.timeoutIntervalForRequest = 30
        config.timeoutIntervalForResource = 60
        return URLSession(configuration: config)
    }
    
    // MARK: - Auth
    func login(email: String, password: String) async throws -> AuthResponse {
        var request = URLRequest(url: URL(string: "\(baseURL)/api/auth/signin")!)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try JSONEncoder().encode(["email": email, "password": password])
        
        let (data, response) = try await session.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 else {
            throw APIError.serverError
        }
        
        let authResponse = try JSONDecoder().decode(AuthResponse.self, from: data)
        accessToken = authResponse.data.accessToken
        refreshToken = authResponse.data.refreshToken
        return authResponse
    }
    
    func signup(email: String, password: String, displayName: String) async throws -> AuthResponse {
        var request = URLRequest(url: URL(string: "\(baseURL)/api/auth/signup")!)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try JSONEncoder().encode(["email": email, "password": password, "displayName": displayName])
        
        let (data, response) = try await session.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 201 else {
            throw APIError.serverError
        }
        
        let authResponse = try JSONDecoder().decode(AuthResponse.self, from: data)
        accessToken = authResponse.data.accessToken
        refreshToken = authResponse.data.refreshToken
        return authResponse
    }
    
    func socialLogin(provider: String, token: String) async throws -> AuthResponse {
        var request = URLRequest(url: URL(string: "\(baseURL)/api/auth/social")!)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try JSONEncoder().encode(["provider": provider, "token": token])
        
        let (data, response) = try await session.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse else { throw APIError.networkError }
        
        if httpResponse.statusCode == 200 {
            let authResponse = try JSONDecoder().decode(AuthResponse.self, from: data)
            accessToken = authResponse.data.accessToken
            refreshToken = authResponse.data.refreshToken
            return authResponse
        }
        
        throw APIError.serverError
    }
    
    func refreshTokens() async throws {
        guard let refreshToken = refreshToken else { throw APIError.authError }
        
        var request = URLRequest(url: URL(string: "\(baseURL)/api/auth/refresh")!)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try JSONEncoder().encode(["refreshToken": refreshToken])
        
        let (data, response) = try await session.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 else {
            throw APIError.authError
        }
        
        let tokens = try JSONDecoder().decode([String: AuthTokens].self, from: data)
        accessToken = tokens["data"]?.accessToken
        refreshToken = tokens["data"]?.refreshToken
    }
    
    func logout() async throws {
        var request = URLRequest(url: URL(string: "\(baseURL)/api/auth/signout")!)
        request.httpMethod = "POST"
        request.allHTTPHeaderFields = authHeader
        
        _ = try await session.data(for: request)
        accessToken = nil
        refreshToken = nil
    }
    
    // MARK: - User
    func getUserProfile() async throws -> User {
        var request = URLRequest(url: URL(string: "\(baseURL)/api/user/profile")!)
        request.allHTTPHeaderFields = authHeader
        
        let (data, response) = try await session.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 else {
            throw APIError.serverError
        }
        
        let wrapped = try JSONDecoder().decode(WrappedResponse<User>.self, from: data)
        return wrapped.data
    }
    
    func getUserStats() async throws -> UserStats {
        var request = URLRequest(url: URL(string: "\(baseURL)/api/user/stats")!)
        request.allHTTPHeaderFields = authHeader
        
        let (data, response) = try await session.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 else {
            throw APIError.serverError
        }
        
        let wrapped = try JSONDecoder().decode(WrappedResponse<UserStats>.self, from: data)
        return wrapped.data
    }
    
    func updateProfile(displayName: String? = nil, avatarUrl: String? = nil) async throws -> User {
        var request = URLRequest(url: URL(string: "\(baseURL)/api/user/profile")!)
        request.httpMethod = "PUT"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.allHTTPHeaderFields = authHeader
        
        var body: [String: String] = [:]
        if let displayName = displayName { body["displayName"] = displayName }
        if let avatarUrl = avatarUrl { body["avatarUrl"] = avatarUrl }
        
        request.httpBody = try JSONEncoder().encode(body)
        
        let (data, response) = try await session.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse, httpResponse.statusCode == 200 else {
            throw APIError.serverError

        let wrapped = try JSONDecoder().decode(WrappedResponse<User>.self, from: data)
```

```swift
struct MiniLineChart: View {
    let  [Double]
    let color: Color
    let height: CGFloat
    
    init(data: [Double], color: Color = Color(hex: "00B4D8"), height: CGFloat = 80) {
        self.data = data
        self.color = color
        self.height = height
    }
    
    var body: some View {
        GeometryReader { geo in
            let maxValue = data.max() ?? 1
            let stepX = geo.size.width / CGFloat(data.count - 1)
            
            Path { path in
                for (index, value) in data.enumerated() {
                    let x = CGFloat(index) * stepX
                    let y = geo.size.height - (value / maxValue) * geo.size.height * 0.8 - geo.size.height * 0.1
                    
                    if index == 0 { path.move(to: CGPoint(x: x, y: y)) }
                    else { path.addLine(to: CGPoint(x: x, y: y)) }
                }
            }
            .stroke(color, style: StrokeStyle(lineWidth: 2.5, lineCap: .round, lineJoin: .round))
            
            Path { path in
                for (index, value) in data.enumerated() {
                    let x = CGFloat(index) * stepX
                    let y = geo.size.height - (value / maxValue) * geo.size.height * 0.8 - geo.size.height * 0.1
                    
                    if index == 0 { path.move(to: CGPoint(x: x, y: y)) }
                    else { path.addLine(to: CGPoint(x: x, y: y)) }
                }
                
                path.addLine(to: CGPoint(x: CGFloat(data.count - 1) * stepX, y: geo.size.height))
                path.addLine(to: CGPoint(x: 0, y: geo.size.height))
                path.closeSubpath()
            }
            .fill(LinearGradient(colors: [color.opacity(0.3), color.opacity(0.02)], startPoint: .top, endPoint: .bottom))
        }
        .frame(height: height)
        .clipShape(.rect(cornerRadius: 12))
    }
}
```

```swift
import Foundation

@MainActor
class StorageManager {
    static let shared = StorageManager()
    
    private let userDefaults = UserDefaults.standard
    
    private enum Keys: String {
        case accessToken
        case refreshToken
        case hasCompletedOnboarding
        case isDarkMode
        case lastScanDate
        case currentStreak
        case cachedUser
        case cachedScans
        case cachedWeeklyTrends
    }
    
    var accessToken: String? {
        get { userDefaults.string(forKey: Keys.accessToken.rawValue) }
        set { userDefaults.set(newValue, forKey: Keys.accessToken.rawValue) }
    }
    
    var refreshToken: String? {
        get { userDefaults.string(forKey: Keys.refreshToken.rawValue) }
        set { userDefaults.set(newValue, forKey: Keys.refreshToken.rawValue) }
    }
    
    var hasCompletedOnboarding: Bool {
        get { userDefaults.bool(forKey: Keys.hasCompletedOnboarding.rawValue) }
        set { userDefaults.set(newValue, forKey: Keys.hasCompletedOnboarding.rawValue) }
    }
    
    var isDarkMode: Bool {
        get { userDefaults.bool(forKey: Keys.isDarkMode.rawValue) }
        set { userDefaults.set(newValue, forKey: Keys.isDarkMode.rawValue) }
    }
    
    var lastScanDate: Date? {
        get { userDefaults.object(forKey: Keys.lastScanDate.rawValue) as? Date }
        set { userDefaults.set(newValue, forKey: Keys.lastScanDate.rawValue) }
    }
    
    var currentStreak: Int {
        get { userDefaults.integer(forKey: Keys.currentStreak.rawValue) }
        set { userDefaults.set(newValue, forKey: Keys.currentStreak.rawValue) }
    }
    
    func saveCachedUser(_ user: User) {
        if let data = try? JSONEncoder().encode(user) {
            userDefaults.set(data, forKey: Keys.cachedUser.rawValue)
        }
    }
    
    func loadCachedUser() -> User? {
        guard let data = userDefaults.data(forKey: Keys.cachedUser.rawValue) else { return nil }
        return try? JSONDecoder().decode(User.self, from: data)
    }
    
    func saveCachedScans(_ scans: [Scan]) {
        if let data = try? JSONEncoder().encode(scans) {
            userDefaults.set(data, forKey: Keys.cachedScans.rawValue)
        }
    }
    
    func loadCachedScans() -> [Scan]? {
        guard let data = userDefaults.data(forKey: Keys.cachedScans.rawValue) else { return nil }
        return try? JSONDecoder().decode([Scan].self, from: data)
    }
    
    func clearAll() {
        accessToken = nil
        refreshToken = nil
        userDefaults.removeObject(forKey: Keys.cachedUser.rawValue)
        userDefaults.removeObject(forKey: Keys.cachedScans.rawValue)
        userDefaults.removeObject(forKey: Keys.cachedWeeklyTrends.rawValue)
    }
}
```

### `AppState.swift`
```swift
import SwiftUI

@Observable
@MainActor
class AppState {
    static let shared = AppState()
    
    var user: User?
    var subscription: Subscription?
    var streak: Streak?
    var scans: [Scan] = []
    var weeklyTrends: [WeeklyTrend] = []
    var missions: [Mission] = []
    var dailyStatus: DailyStatus?
    var weeklySummary: WeeklySummary?
    var isLoading = false
    var errorMessage: String?
    var isAuthenticated = false
    var hasCompletedOnboarding = false
    var scanCountToday = 0
    var maxFreeScans = 3
    
    init() { loadState() }
    
    func loadState() {
        hasCompletedOnboarding = StorageManager.shared.hasCompletedOnboarding
        ThemeManager.shared.isDarkMode = StorageManager.shared.isDarkMode
        
        if let token = StorageManager.shared.accessToken {
            APIService.shared.accessToken = token
            isAuthenticated = true
            Task { await loadUserData() }
        }
    }
    
    func loadUserData() async {
        isLoading = true
        defer { isLoading = false }
        
        do {
            async let userProfile = APIService.shared.getUserProfile()
            async let userStats = APIService.shared.getUserStats()
            async let userScans = APIService.shared.getUserScans(limit: 10)
            async let trends = APIService.shared.getWeeklyTrends()
            async let dailyStat = APIService.shared.getDailyStatus()
            async let weeklySum = APIService.shared.getWeeklySummary()
            async let subs = APIService.shared.getSubscription()
            async let userMissions = APIService.shared.getMissions()
            
            let (profile, stats, recentScans, trendsData, status, summary, subscription, missionsData) = try await (
                userProfile, userStats, userScans, trends, dailyStat, weeklySum, subs, userMissions
            )
            
            user = profile
            scans = recentScans
            weeklyTrends = trendsData
            dailyStatus = status
            weeklySummary = summary
            subscription = subscription
            missions = missionsData
            scanCountToday = recentScans.filter { Calendar.current.isDateInToday($0.scannedAt) }.count
            
            StorageManager.shared.saveCachedUser(profile)
            StorageManager.shared.saveCachedScans(recentScans)
        } catch {
            errorMessage = error.localizedDescription
            print("Load user data error: \(error)")
        }
    }
    
    func refreshData() async {
        await loadUserData()
    }
    
    func authenticate(email: String, password: String) async throws {
        let response = try await APIService.shared.login(email: email, password: password)
        StorageManager.shared.accessToken = response.data.accessToken
        StorageManager.shared.refreshToken = response.data.refreshToken
        user = response.data.user
        isAuthenticated = true
        await loadUserData()
    }
    
    func register(email: String, password: String, displayName: String) async throws {
        let response = try await APIService.shared.signup(email: email, password: password, displayName: displayName)
        StorageManager.shared.accessToken = response.data.accessToken
        StorageManager.shared.refreshToken = response.data.refreshToken
        user = response.data.user
        isAuthenticated = true
        await loadUserData()
    }
    
    func socialAuthenticate(provider: String, token: String) async throws {
        let response = try await APIService.shared.socialLogin(provider: provider, token: token)
        StorageManager.shared.accessToken = response.data.accessToken
        StorageManager.shared.refreshToken = response.data.refreshToken
        user = response.data.user
        isAuthenticated = true
        await loadUserData()
    }
    
    func logout() async {
        try? await APIService.shared.logout()
        StorageManager.shared.clearAll()
        
        user = nil
        subscription = nil
        scans = []
        weeklyTrends = []
        missions = []
        dailyStatus = nil
        weeklySummary = nil
        isAuthenticated = false
    }
    
    func completeOnboarding() {
        hasCompletedOnboarding = true
        StorageManager.shared.hasCompletedOnboarding = true
    }
    
    func canPerformScan() -> Bool {
        guard let subscription = subscription else { return false }
        if subscription.tier == .free {
            return scanCountToday < maxFreeScans
        }
        return true
    }
    
    var remainingFreeScans: Int {
        guard let subscription = subscription, subscription.tier == .free else { return .max }
        return max(0, maxFreeScans - scanCountToday)
    }
    
    var weeklyAverage: Int {
        guard !weeklyTrends.isEmpty else { return 0 }
        let avg = weeklyTrends.reduce(0) { $0 + $1.averageScore } / Double(weeklyTrends.count)
        return Int(avg)
    }
    
    var totalScans: Int {
        user != nil ? (scans.count + Int.random(in: 15...30)) : 0
    }
    
    var streakDays: Int {
        streak?.scanningStreak ?? currentStreak
    }
}
```

async function retryWithBackoff<T>(
  fn: () => Promise<T>,
  maxRetries = 3,
  baseDelay = 1000
): Promise<T> {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error: any) {
      if (attempt === maxRetries) throw error;
      if (error.status === 429 || error.status >= 500) {
        const delay = baseDelay * Math.pow(2, attempt - 1) + Math.random() * 1000;
        logger.warn(`AI call failed (attempt ${attempt}), retrying in ${delay}ms...`);
        await new Promise((resolve) => setTimeout(resolve, delay));
      } else {
        throw error; // don't retry 4xx errors
      }
    }
  }
  throw new Error('Max retries exceeded');
}

interface UserStage {

  scansCount: number;

interface InsightPayload {
  userId: string;
  scanId: string;
  detectedProduct: string;
  sugarGrams: number;
  category: string;

export class InsightEngine {
  /**
   * Determine the user's current stage based on scan history.
   */
  static async getUserStage(userId: string): Promise<UserStage> {
    const firstScan = await prisma.scan.findFirst({
      where: { userId },
      orderBy: { createdAt: 'asc' },
      select: { createdAt: true },
    });

    const totalScans = await prisma.scan.count({ where: { userId } });

    if (!firstScan) {
      return { stage: 'week1', scansCount: 0, daysSinceFirstScan: 0 };
    }

    const daysSinceFirstScan = Math.floor((Date.now() - firstScan.createdAt.getTime()) / (1000 * 60 * 60 * 24));

    if (daysSinceFirstScan <= 7) {
      return { stage: 'week1', scansCount: totalScans, daysSinceFirstScan };
    }
    if (daysSinceFirstScan <= 14) {
      return { stage: 'week2', scansCount: totalScans, daysSinceFirstScan };
    }
    return { stage: 'veteran', scansCount: totalScans, daysSinceFirstScan };
  }

  /**
   * Generate contextual insights based on stage and recent scans.
   */
  static async generateInsights(payload: InsightPayload): Promise<string[]> {
    const insights: string[] = [];
    const { userId, detectedProduct, sugarGrams, category, scanId } = payload;

    const userStage = await this.getUserStage(userId);
    const recentScans = await prisma.scan.findMany({
      where: { userId },
      orderBy: { scannedAt: 'desc' },
      take: 20,
      include: { composition: true },
    });

    // Base insight: hydration impact
    const hydrationLevel = sugarGrams > 20 ? 'dehydrating' : 'neutral';
    if (hydrationLevel === 'dehydrating') {
      insights.push(`This ${detectedProduct} has ${sugarGrams}g of sugar, which can cause a temporary dehydration effect.`);
    }

    // Stage-specific insights
    switch (userStage.stage) {
      case 'week1':
        // Simple awareness
        insights.push(`You've scanned ${userStage.scansCount} drinks so far. Keep tracking to discover patterns!`);
        break;
      case 'week2':
        // Frequency warnings
        const sameProductCount = recentScans.filter(s => s.detectedProduct === detectedProduct).length;
        if (sameProductCount >= 3) {
          insights.push(`This is your ${sameProductCount}th ${detectedProduct} in the last ${recentScans.length} scans. Consider varying your choices.`);
        }
        break;
      case 'veteran':
        // Trend analysis
        const weeklySugarDrop = await this.getWeeklySugarChange(userId);
        if (weeklySugarDrop !== null && weeklySugarDrop < -20) {
          insights.push(`Your average sugar intake per scan dropped ${Math.abs(weeklySugarDrop)}% compared to last week. Great improvement!`);
        } else if (weeklySugarDrop !== null && weeklySugarDrop > 10) {
          insights.push(`Your sugar intake increased by ${Math.round(weeklySugarDrop)}% this week. Try to cut back on high-sugar drinks.`);
        }

        // Energy crash frequency
        const energyCrashRatio = await this.getEnergyCrashRatio(userId);
        if (energyCrashRatio > 0.4) {
          insights.push(`You've reported energy crashes after ${Math.round(energyCrashRatio * 100)}% of scans. Drinks with caffeine > 100mg may be contributing.`);
        }

        // Drink quality decline
        const qualityTrend = await this.getConsistencyTrend(userId);
        if (qualityTrend < -0.2) {
          insights.push(`Your drink quality score has been declining. Try to include more low-sugar options.`);
        }
        break;
    }

    // Always add a positive tip
    if (sugarGrams < 5) {
      insights.push(`Great choice! This drink is low in sugar, supporting stable energy levels.`);
    }

    return insights;
  }

  private static async getWeeklySugarChange(userId: string): Promise<number | null> {
    const oneWeekAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000);
    const twoWeeksAgo = new Date(Date.now() - 14 * 24 * 60 * 60 * 1000);

    const [recentAvg, previousAvg] = await Promise.all([
      prisma.composition.aggregate({
        where: { scan: { userId, scannedAt: { gte: oneWeekAgo } } },
        _avg: { sugarGrams: true },
      }),
      prisma.composition.aggregate({
        where: { scan: { userId, scannedAt: { gte: twoWeeksAgo, lt: oneWeekAgo } } },
        _avg: { sugarGrams: true },
      }),
    ]);

    if (!recentAvg._avg.sugarGrams || !previousAvg._avg.sugarGrams) return null;

    return ((recentAvg._avg.sugarGrams - previousAvg._avg.sugarGrams) / previousAvg._avg.sugarGrams) * 100;
  }

    return 0.2;
  }

  private static async getConsistencyTrend(userId: string): Promise<number> {
    // Could calculate based on standard deviation of impact scores
    const scores = await prisma.scan.findMany({
      where: { userId },
      orderBy: { scannedAt: 'asc' },
      select: { impactScore: true },
    });
    if (scores.length < 5) return 0;
    const recentScores = scores.slice(-10).map(s => s.impactScore);
    const olderScores = scores.slice(-20, -10).map(s => s.impactScore);
    const recentAvg = recentScores.reduce((a, b) => a + b, 0) / recentScores.length;
    const olderAvg = olderScores.length ? olderScores.reduce((a, b) => a + b, 0) / olderScores.length : recentAvg;
    return (recentAvg - olderAvg) / olderAvg;
  }

// Helper: calculate impact score 0-100 based on composition
function calculateImpactScore(composition: any): number {
  let score = 100;
  // Penalize high sugar
  if (composition.sugarGrams > 30) score -= 25;
  else if (composition.sugarGrams > 15) score -= 15;
  else if (composition.sugarGrams > 5) score -= 5;

  // Penalize high caffeine
  if (composition.caffeineMg > 200) score -= 15;
  else if (composition.caffeineMg > 100) score -= 8;

  // Penalize artificial sweeteners
  if (composition.artificialSweeteners) score -= 10;

  // Penalize many additives
  if (composition.additives && composition.additives.length > 5) score -= 10;
  else if (composition.additives && composition.additives.length > 2) score -= 5;

  // Reward protein / low fat
  if (composition.proteinGrams > 5) score += 5;
  if (composition.fatGrams < 2) score += 5;

  return Math.max(0, Math.min(100, score));
}

// Helper: simple hydration level 0-100 (higher = better)
function calculateHydration(sugarGrams: number, caffeineMg: number): number {
  let level = 80; // baseline
  if (sugarGrams > 20) level -= 25;
  else if (sugarGrams > 10) level -= 10;
  if (caffeineMg > 150) level -= 15;
  else if (caffeineMg > 75) level -= 5;
  return Math.max(0, level);
}

export class ScanOrchestrator {
  /**
   * Main pipeline: save scan → AI analysis → insights → streak → missions → analytics.
   * AI analysis runs synchronously for immediate response; background job handles the rest.
   */
  static async processScan(input: ProcessScanInput) {
    logger.info(`Orchestrator: starting scan for user ${input.userId}`);

    // 1. Validate scan limit for free users
    const userSubscription = await prisma.subscription.findUnique({ where: { userId: input.userId } });
    const isFree = !userSubscription || userSubscription.tier === 'FREE';
    if (isFree) {
      const todayStart = new Date();
      todayStart.setHours(0, 0, 0, 0);
      const todayScans = await prisma.scan.count({
        where: { userId: input.userId, createdAt: { gte: todayStart } },
      });
      if (todayScans >= 3) {
        logger.warn(`Free user ${input.userId} reached daily scan limit`);
        throw new Error('FREE_LIMIT_REACHED');
      }
    }

    // 2. Call AI service to analyze image
    const aiResult = await AIService.analyzeDrinkImage(input.imageUrl);

    // 3. Calculate impact score and hydration
    const impactScore = calculateImpactScore(aiResult.composition);
    const hydrationLevel = calculateHydration(
      aiResult.composition.sugarGrams,
      aiResult.composition.caffeineMg
    );

    // 4. Determine glycemic impact
    const glycemicImpact: 'LOW' | 'MEDIUM' | 'HIGH' =
      aiResult.composition.sugarGrams > 20 ? 'HIGH' :
      aiResult.composition.sugarGrams > 10 ? 'MEDIUM' : 'LOW';

    // 5. Create Scan with composition and ingredients in a transaction
    const scan = await prisma.$transaction(async (tx) => {
      // Create composition
      const composition = await tx.composition.create({
        data: {
          calories: aiResult.composition.calories,
          sugarGrams: aiResult.composition.sugarGrams,
          caffeineMg: aiResult.composition.caffeineMg,
          sodiumMg: aiResult.composition.sodiumMg,
          fatGrams: aiResult.composition.fatGrams,
          proteinGrams: aiResult.composition.proteinGrams,
          additives: aiResult.composition.additives,
          artificialSweeteners: aiResult.composition.artificialSweeteners,
          servingSize: aiResult.composition.servingSize,
          servingUnit: aiResult.composition.servingUnit,
          ingredients: {
            create: aiResult.composition.ingredients.map((ing: any, idx: number) => ({
              name: ing.name,
              function: ing.function,
              healthRole: ing.healthRole,
              riskLevel: ing.riskLevel,
              description: ing.description,
              position: ing.position ?? idx,
            })),
          },
        },
        include: { ingredients: true },
      });

      // Create report
      const report = await tx.report.create({
        data: {
          shortTermEnergy: aiResult.report.shortTermEnergy,
          shortTermBloodSugar: aiResult.report.shortTermBloodSugar,
          shortTermBodyReaction: aiResult.report.shortTermBodyReaction,
          shortTermHydration: aiResult.report.shortTermHydration,
          mediumEnergyStability: aiResult.report.mediumEnergyStability,
          mediumPhysicalChanges: aiResult.report.mediumPhysicalChanges,
          mediumHabitRisk: aiResult.report.mediumHabitRisk,
          mediumSleepQuality: aiResult.report.mediumSleepQuality,
          longTermHealthTrend: aiResult.report.longTermHealthTrend,
          longTermMetabolic: aiResult.report.longTermMetabolic,
          longTermRiskAccumulation: aiResult.report.longTermRiskAccumulation,
          longTermNutritional: aiResult.report.longTermNutritional,
          viralStatement: aiResult.report.viralStatement,
          tiktokHook: aiResult.report.tiktokHook,
        },
      });

      // Create AI insight record
      const aiInsight = await tx.aIInsightRecord.create({
        data: {
          insight: aiResult.aiInsight,
          summary: aiResult.aiSummary,
        },
      });

      // Create the scan
      const scan = await tx.scan.create({
        data: {
          userId: input.userId,
          imageUrl: input.imageUrl,
          detectedProduct: aiResult.detectedProduct,
          brand: aiResult.brand,
          category: aiResult.category,
          confidenceScore: 0.95,
          impactScore,
          hydrationLevel,
          glycemicImpact,
          compositionId: composition.id,
          reportId: report.id,
          aiInsightId: aiInsight.id,
          scannedAt: new Date(),
        },
        include: {
          composition: { include: { ingredients: true } },
          report: true,
          aiInsight: true,
        },
      });

      return scan;
    });

    // 6. Enqueue background job for insights, streak, missions (non‑blocking)
    await addScanToQueue({
      scanId: scan.id,
      userId: input.userId,
      detectedProduct: scan.detectedProduct,
      sugarGrams: scan.composition?.sugarGrams ?? 0,
      category: scan.category,
    });

    logger.info(`Orchestrator: scan ${scan.id} created for user ${input.userId}`);
    return scan;
  }
}

const scanQueue = new Queue<ScanJobData>('scan-processing', {
  redis: { host: process.env.REDIS_HOST || 'localhost', port: Number(process.env.REDIS_PORT) || 6379 },
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: 'exponential', delay: 2000 },
    removeOnComplete: 100,
    removeOnFail: 50,
  },
});

// Process jobs
scanQueue.process(async (job) => {
  const { scanId, userId, detectedProduct, sugarGrams, category } = job.data;
  logger.info(`Queue: processing scan ${scanId}`);

  try {
    // 1. Generate contextual insights
    const insights = await InsightEngine.generateInsights({
      userId,
      scanId,
      detectedProduct,
      sugarGrams,
      category,
    });

    // If we had an insights model, we would save them here (e.g., add to a table or update scan)
    // For now, log them
    if (insights.length > 0) {
      logger.info(`Insights for scan ${scanId}: ${insights.join(' | ')}`);
    }

    // 2. Update streak (already handled by StreakService on scan creation, but we can do additional checks)
    await StreakService.checkAndUpdateStreak(userId);

    // 3. Check missions (progress update)
    await MissionService.checkScanRelatedMissions(userId, scanId, detectedProduct);

    // 4. Update analytics (placeholder – could aggregate stats)
    logger.info(`Queue: scan ${scanId} background processing complete`);
  } catch (error: any) {
    logger.error(`Queue: error processing scan ${scanId}: ${error.message}`);
    throw error; // Bull will retry
  }
});

// Helper to add a job
export async function addScanToQueue(data: ScanJobData): Promise<void> {
  await scanQueue.add(data, {
    delay: 0,
    priority: 1,
  });
  logger.info(`Queue: job added for scan ${data.scanId}`);
}

export class StreakService {
  /**
   * After a new scan, check and update the user's streak.
   * A streak is maintained if scans occur on consecutive calendar days.
   */
  static async checkAndUpdateStreak(userId: string): Promise<void> {
    try {
      const user = await prisma.user.findUnique({
        where: { id: userId },
        select: { streak: true, lastScanDate: true },
      });

      if (!user) return;

      const today = new Date();
      today.setHours(0, 0, 0, 0);
      const yesterday = new Date(today);
      yesterday.setDate(yesterday.getDate() - 1);

      let newStreak = user.streak ?? 0;

      if (!user.lastScanDate) {
        // First scan ever
        newStreak = 1;
      } else {
        const lastScan = new Date(user.lastScanDate);
        lastScan.setHours(0, 0, 0, 0);

        if (lastScan.getTime() === today.getTime()) {
          // Already scanned today – no streak change
          return;
        } else if (lastScan.getTime() === yesterday.getTime()) {
          // Consecutive day
          newStreak += 1;
        } else {
          // Missed a day – reset
          newStreak = 1;
        }
      }

      await prisma.user.update({
        where: { id: userId },
        data: {
          streak: newStreak,
          lastScanDate: today,
        },
      });

      logger.info(`Streak updated for user ${userId}: ${newStreak} days`);
    } catch (error: any) {
      logger.error(`Streak update failed for user ${userId}: ${error.message}`);
    }
  }
}

          case 'WATER_FIRST':
            // Check if first scan of the day is water
            break;

          case 'CLEAN_DRINK':
            // Check for no additives
            break;

type RequiredTier = 'STARTER' | 'PRO' | 'ELITE' | 'FAMILY';

/**
 * 中间件工厂：要求用户订阅至少达到指定 tier。
 * 例如 `requireSubscription('PRO')` 会阻止 Free 和 Starter 用户。
 */
export function requireSubscription(minTier: RequiredTier) {
  const tierHierarchy: Record<string, number> = {
    FREE: 0,
    STARTER: 1,
    PRO: 2,
    ELITE: 3,
    FAMILY: 4,
  };

  return async (req: Request, res: Response, next: NextFunction) => {
    try {
      const userId = req.user!.id;

      const subscription = await prisma.subscription.findUnique({
        where: { userId },
      });

      if (!subscription || subscription.status !== 'ACTIVE') {
        return res.status(402).json({
          error: 'Subscription required',
          upgradeRequired: true,
          message: 'This feature requires an active subscription.',
        });
      }

      const userTier = tierHierarchy[subscription.tier] ?? 0;
      const requiredTier = tierHierarchy[minTier] ?? 0;

      if (userTier < requiredTier) {
        return res.status(402).json({
          error: 'Upgrade required',
          upgradeRequired: true,
          message: `This feature requires the ${minTier} plan or higher.`,
          requiredTier: minTier,
          currentTier: subscription.tier,
        });
      }

      next();
    } catch (error: any) {
      logger.error(`Subscription middleware error: ${error.message}`);
      return res.status(500).json({ error: 'Internal server error' });
    }
  };
}
